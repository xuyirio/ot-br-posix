183 template <typename ValueType>
184 otbrError SignalPropertyChanged(const std::string &aInterfaceName,
185 const std::string &aPropertyName,
186 const ValueType &aValue)
187 {
188 UniqueDBusMessage signalMsg = NewSignalMessage(DBUS_INTERFACE_PROPERTIES, DBUS_PROPERTIES_CHANGED_SIGNAL);
189 DBusMessageIter iter, subIter, dictEntryIter;
190 otbrError error = OTBR_ERROR_NONE;
191
192 VerifyOrExit(signalMsg != nullptr, error = OTBR_ERROR_DBUS);
193 dbus_message_iter_init_append(signalMsg.get(), &iter);
194
195 // interface_name
196 VerifyOrExit(DBusMessageEncode(&iter, aInterfaceName) == OTBR_ERROR_NONE, error = OTBR_ERROR_DBUS);
197
198 // changed_properties
199 VerifyOrExit(dbus_message_iter_open_container(&iter, DBUS_TYPE_ARRAY,
200 "{" DBUS_TYPE_STRING_AS_STRING DBUS_TYPE_VARIANT_AS_STRING "}",
201 &subIter),
202 error = OTBR_ERROR_DBUS);
203 VerifyOrExit(dbus_message_iter_open_container(&subIter, DBUS_TYPE_DICT_ENTRY, nullptr, &dictEntryIter),
204 error = OTBR_ERROR_DBUS);
205
206 SuccessOrExit(error = DBusMessageEncode(&dictEntryIter, aPropertyName));
207 SuccessOrExit(error = DBusMessageEncodeToVariant(&dictEntryIter, aValue));
208
209 VerifyOrExit(dbus_message_iter_close_container(&subIter, &dictEntryIter), error = OTBR_ERROR_DBUS);
210 VerifyOrExit(dbus_message_iter_close_container(&iter, &subIter), error = OTBR_ERROR_DBUS);
211
212 // invalidated_properties
213 SuccessOrExit(error = DBusMessageEncode(&iter, std::vector<std::string>()));
214
215 if (otbrLogGetLevel() >= OTBR_LOG_DEBUG)
216 {
217 otbrLogDebug("Signal %s.%s", aInterfaceName.c_str(), aPropertyName.c_str());
218 DumpDBusMessage(*signalMsg);
219 }
220
221 VerifyOrExit(dbus_connection_send(mConnection, signalMsg.get(), nullptr), error = OTBR_ERROR_DBUS);
222
223 exit:
224 return error;
225 }
